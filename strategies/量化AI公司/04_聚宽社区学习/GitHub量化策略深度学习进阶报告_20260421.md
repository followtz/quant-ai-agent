            return {'action': 'BUY_VOLATILITY', 'conviction': 0.9}

class TurtleTrading:
    """海龟交易系统"""
    
    def __init__(self, 
                 entry_period=20,      # 入场周期
                 exit_period=10,       # 出场周期
                 atr_period=20,        # ATR周期
                 risk_fraction=0.02):  # 风险比例
    
    def calculate_units(self, price, atr, account_size):
        """计算交易单位"""
        unit_size = (account_size * risk_fraction) / atr
        return int(unit_size)
    
    def check_entry(self, current_price, high_n, low_n):
        """检查是否触发入场"""
        if current_price > high_n:
            return 'LONG'
        elif current_price < low_n:
            return 'SHORT'
        return None
    
    def check_exit(self, current_price, entry_price, low_n, high_n, profit_target):
        """检查是否触发出场"""
        if profit_target and (current_price - entry_price) >= profit_target:
            return 'TAKE_PROFIT'
        if low_n < entry_price - 2 * (entry_price - low_n):
            return 'STOP_LOSS'
        return None

class StockTradingEnv(gym.Env):
    """股票交易强化学习环境"""
    
    def __init__(self, 
                 initial_balance=10000,
                 transaction_cost=0.001,
                 tech_indicator_list=['macd', 'rsi', 'cci', 'dx']):
        super().__init__()
        self.initial_balance = initial_balance
        self.transaction_cost = transaction_cost
        self.tech_indicator_list = tech_indicator_list
        
        # 动作空间：-1(卖) 0(持有) 1(买)
        self.action_space = spaces.Discrete(3)
        
        # 观察空间：账户状态 + 技术指标
        self.observation_space = spaces.Box(
            low=-np.inf, high=np.inf, 
            shape=(len(tech_indicator_list) + 4,)
        )
    
    def _calculate_portfolio_state(self):
        """计算当前组合状态"""
        return np.array([
            self.balance / self.initial_balance,                    # 现金比例
            self.total_asset / self.initial_balance,              # 资产比例
            self.position_value / self.total_asset if self.total_asset > 0 else 0,  # 持仓比例
            self.stock_holding / self.stock_max_holding,         # 持仓占比
            *[self.tech_indicators[t][self.current_step] 
              for t in self.tech_indicator_list]                  # 技术指标
        ])
    
    def step(self, action):
        """执行一步交易"""
        # 执行动作
        if action == 1:  # 买入
            self._buy()
        elif action == 0:  # 持有
            pass
        else:  # 卖出
            self._sell()
        
        # 计算奖励（基于收益）
        reward = self._calculate_return()
        
        # 检查终止条件
        done = self.current_step >= self.max_steps
        
        return self._calculate_portfolio_state(), reward, done, {}
    
    def _calculate_return(self):
        """计算收益率作为奖励"""
        return (self.total_asset - self.initial_balance) / self.initial_balance

#### 1.2.3 PPO算法实现（ppo.py核心）

```python
class PPOAgent:
    """Proximal Policy Optimization 智能体"""
    
    def __init__(self, 
                 state_dim,
                 action_dim,
                 lr_actor=3e-4,
                 lr_critic=1e-3,
                 clip_ratio=0.2,
                 gamma=0.99,
                 lam=0.95):
        
        self.gamma = gamma          # 折扣因子
        self.lam = lam              # GAE参数
        self.clip_ratio = clip_ratio # 裁剪比例
        
        # Actor网络（策略网络）
        self.actor = self._build_network(
            input_dim=state_dim,
            output_dim=action_dim,
            lr=lr_actor,
            output_activation='softmax'
        )
        
        # Critic网络（价值网络）
        self.critic = self._build_network(
            input_dim=state_dim,
            output_dim=1,
            lr=lr_critic,
            output_activation='linear'
        )
    
    def update(self, trajectory):
        """
        使用PPO算法更新策略
        关键：使用clipped surrogate objective防止策略更新过大
        """
        states, actions, rewards, old_log_probs, values = trajectory
        
        # 计算优势函数（GAE）
        advantages = self._compute_gae(rewards, values)
        
        # 标准化优势
        advantages = (advantages - advantages.mean()) / (advantages.std() + 1e-8)
        
        # PPO更新（多次epoch）
        for _ in range(10):
            new_log_probs = self.actor.get_log_prob(states, actions)
            ratio = torch.exp(new_log_probs - old_log_probs)
            
            # Clipped objective
            surr1 = ratio * advantages
            surr2 = torch.clamp(ratio, 1-self.clip_ratio, 1+self.clip_ratio) * advantages
            
            # 策略损失（最小化）
            policy_loss = -torch.min(surr1, surr2).mean()
            
            # 价值损失
            value_loss = F.mse_loss(self.critic(states), values + advantages)
            
            # 联合优化
            self.optimizer.zero_grad()
            (policy_loss + 0.5 * value_loss).backward()
            self.optimizer.step()
```

                    ppo_action=action,
                    ml_action=ml_action,
                    ppo_weight=ppo_weight,
                    ml_weight=ml_weight
                )

                next_state, reward, done, info = self.env.step(final_action)
                
                # 存储经验
                self.ppo_agent.store_transition(
                    state, action, reward, 
                    self.ppo_agent.get_log_prob(state, action),
                    self.ppo_agent.critic(state)
                )
                
                state = next_state
                episode_reward += reward
            
            # 周期性更新PPO
            if episode % 10 == 0:
                self.ppo_agent.update(self.ppo_agent.buffer)
    
    def predict(self, current_state):

        ml_signal = self.ml_model.predict(current_state)
        
        # 融合概率

        return {
            'action': np.argmax(fused_probs),
            'confidence': np.max(fused_probs),
            'ppo_prob': ppo_probs,
            'ml_prob': ml_signal['prob'],
            'fused_prob': fused_probs

class MultiFactorModel:
    """多因子选股模型"""
    
    def __init__(self, factors_config):
        self.factors = factors_config
        self.model = self._build_ensemble_model()
    
    def calculate_factors(self, stock_data):
        """计算所有因子"""
        factors = {}
        
        # 价值因子
        factors['pe_ratio'] = stock_data['price'] / stock_data['earnings_per_share']
        factors['pb_ratio'] = stock_data['price'] / stock_data['book_value']
        factors['pcf_ratio'] = stock_data['price'] / stock_data['cash_flow']
        
        # 动量因子
        factors['momentum_1m'] = stock_data['price'] / stock_data['price'].shift(20) - 1
        factors['momentum_3m'] = stock_data['price'] / stock_data['price'].shift(60) - 1
        factors['momentum_6m'] = stock_data['price'] / stock_data['price'].shift(120) - 1
        
        # 质量因子
        factors['roe'] = stock_data['net_income'] / stock_data['equity']
        factors['debt_to_equity'] = stock_data['total_debt'] / stock_data['equity']
        factors['gross_margin'] = stock_data['gross_profit'] / stock_data['revenue']
        
        # 波动率因子
        factors['volatility_20d'] = stock_data['returns'].rolling(20).std()
        factors['volatility_60d'] = stock_data['returns'].rolling(60).std()
        
        return factors
    
    def calculate_composite_score(self, factors):
        """
        计算综合评分
        使用Z-score标准化 + 因子加权
        """
        # 标准化
        z_scores = {}
        for factor, values in factors.items():
            z_scores[factor] = (values - values.mean()) / values.std()
        
        # 加权求和
        weights = {
            'pe_ratio': -0.1,      # 低PE更好（负权重）
            'momentum_1m': 0.15,
            'momentum_3m': 0.15,
            'roe': 0.2,
            'volatility_20d': 0.1,
            'debt_to_equity': -0.1  # 低负债更好（负权重）
        }
        
        composite_score = sum(
            z_scores[f] * weights.get(f, 0) 
            for f in z_scores
        )
        
        return composite_score
    
    def rank_stocks(self, stocks_data):
        """对股票进行排名"""
        factors = self.calculate_factors(stocks_data)
        scores = self.calculate_composite_score(factors)
        
        # 返回排名
        return stocks_data.assign(
            composite_score=scores
        ).sort_values('composite_score', ascending=False)
